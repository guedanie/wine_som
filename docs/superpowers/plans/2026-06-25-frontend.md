# Terroir Frontend Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the Terroir React SPA — 4 screens wired to the existing FastAPI backend — using the high-fidelity prototype in `frontend/design-system/` as the visual reference.

**Architecture:** Vite 5 SPA with React 18 and React Router v6. Shared UI components are ported from `frontend/design-system/ui_kits/wine-app/shared.jsx`. Screen components follow the same layouts as their prototype counterparts in `PreferenceCapture.jsx`, `ChatRecommend.jsx`, `RegionDossier.jsx`, and `Discovery.jsx`. Data fetching is plain `fetch` via `lib/api.js`.

**Tech Stack:** Vite 5, React 18, React Router v6, Tailwind v3, Vitest, @testing-library/react, native fetch.

## Global Constraints

- All commands run from `frontend/` unless stated otherwise
- Backend runs at `http://localhost:8000`; frontend dev server at `http://localhost:5173`
- CSS custom properties from `design-system/colors_and_type.css` are the source of truth — never hardcode hex values in component files
- Serif font (DM Serif Display) must not appear below 23px
- Sharp corners (`border-radius: 0`) on all frames, cards, buttons, tags — soft radius (`8px`) on chat bubbles only
- Contour map SVG only on RegionDossier page, never on WineCard grids
- No auth, no Supabase JS client in the frontend
- `VITE_API_URL` env var points to the backend (default: `http://localhost:8000`)
- Python 3.9 backend — no frontend changes affect this, but keep in mind when reading backend code

// frontend/src/lib/api.js
const BASE = import.meta.env?.VITE_API_URL ?? 'http://localhost:8000';

export async function recommend(req) {
  const res = await fetch(`${BASE}/api/recommend`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(req),
  });
  if (!res.ok) {
    const body = await res.json().catch(() => ({}));
    throw new Error(body.detail ?? `HTTP ${res.status}`);
  }
  return res.json();
}

export async function getWine(id) {
  const res = await fetch(`${BASE}/api/wines/${id}`);
  if (!res.ok) throw new Error(`HTTP ${res.status}`);
  return res.json();
}

// frontend/src/components/Contours.jsx
function tRing(cx, cy, R, wob, seed, sx, pts = 80) {
  let d = '';
  for (let j = 0; j <= pts; j++) {
    const t = (j / pts) * Math.PI * 2;
    const rr = R + Math.sin(t * 3 + seed) * wob * (0.6 + 0.4 * Math.sin(t * 2 + seed));
    d += (j === 0 ? 'M' : 'L') +
      (cx + Math.cos(t) * rr * sx).toFixed(1) + ' ' +
      (cy + Math.sin(t) * rr).toFixed(1);
  }
  return d + 'Z';
}

export default function Contours({ w = 600, h = 120, color = '#B08D57', cfg }) {
  const c = cfg ?? { cx: w / 2, cy: h / 2, r0: 10, step: 9, count: 9, wob: 7, seed: 1.4, sx: 1.6 };
  const paths = [];
  for (let i = 0; i < c.count; i++) {
    paths.push(tRing(c.cx, c.cy, c.r0 + i * c.step, c.wob, c.seed + i * 0.5, c.sx));
  }
  return (
    <svg viewBox={`0 0 ${w} ${h}`} preserveAspectRatio="xMidYMid slice"
      style={{ position: 'absolute', inset: 0, width: '100%', height: '100%' }}>
      {paths.map((d, i) => (
        <path key={i} d={d} fill="none" stroke={color}
          strokeWidth={i === c.count - 1 ? 1.6 : 1}
          opacity={0.3 + (i / c.count) * 0.6} />
      ))}
      <circle cx={c.cx} cy={c.cy} r="3" fill={color} />
    </svg>
  );
}

export default function ChatRecommend() {
  const { state }  = useLocation();
  const navigate   = useNavigate();
  const { prefs, apiReq } = state ?? {};

  const [messages, setMessages] = useState([]);
  const [picks,    setPicks]    = useState([]);
  const [loading,  setLoading]  = useState(true);
  const [error,    setError]    = useState(null);
  const [input,    setInput]    = useState('');

  // All hooks must be called before any early return

    if (!prefs) return;
    setMessages([{ role: 'user', text: prefs.styles.join(', ') + ' · under $' + prefs.budget + ' · ' + prefs.occasion.toLowerCase() }]);

  }, []);

    setLoading(true);
    setError(null);

      const data = await recommend(req);
      setMessages(prev => [...prev, { role: 'sommelier', text: data.narrative }]);
      setPicks(data.picks.map(deriveWineCardMeta));
    } catch (err) {
      setError(err.message);
    } finally {
      setLoading(false);

              : <SommelierBubble key={i}>{m.text}</SommelierBubble>
          )}
          {loading && <SommelierBubble>Finding the right bottles for you…</SommelierBubble>}
          {error && (

              <div>{error}</div>

                style={{ cursor: 'pointer', fontFamily: 'var(--font-sans)', fontSize: 12, color: 'var(--bordeaux)', background: 'var(--bordeaux-tint)', border: 'none', borderRadius: 999, padding: '6px 12px' }}>
                {f}
              </button>

              value={input}
              onChange={e => setInput(e.target.value)}
              onKeyDown={e => { if (e.key === 'Enter' && input.trim()) { handleFollowup(input.trim()); setInput(''); } }}
              placeholder="Ask a follow-up…"
              style={{ flex: 1, border: 'none', background: 'transparent', outline: 'none', fontFamily: 'var(--font-sans)', fontSize: 14, color: 'var(--ink)', padding: '11px 13px' }}

            </button>

          <>
            <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'baseline', marginBottom: 18 }}>
