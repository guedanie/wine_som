# Terroir — Supabase-First Foundation Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the complete data foundation for Terroir — Supabase schema, CSV seed ingestion, GrapeMinds enrichment pipeline, retail scrapers, and Claude recommendation engine — before touching any frontend work.

**Architecture:** Supabase as the unified backend (PostgreSQL + Auth + RLS + pg_cron). FastAPI as a thin Python layer for scraping orchestration, enrichment pipeline, and Claude API calls. No Celery, no Redis, no Alembic — Supabase CLI manages migrations as plain SQL files.

**Tech Stack:** Python 3.12, FastAPI, Supabase CLI, supabase-py, Playwright, curl (via subprocess for GrapeMinds), Anthropic SDK, Docker (local Supabase only)

```python
# backend/scripts/seed_from_csv.py
import csv
import sys
from dataclasses import dataclass
from datetime import datetime
from pathlib import Path
from db import get_service_client

RED_VARIETALS = {"cabernet sauvignon","merlot","pinot noir","syrah","shiraz",
                 "malbec","zinfandel","sangiovese","tempranillo","grenache","red blend"}
WHITE_VARIETALS = {"chardonnay","sauvignon blanc","pinot grigio","pinot gris",
                   "riesling","albariño","viognier","white blend","moscato"}
SPARKLING_TERMS = {"prosecco","champagne","cava","sparkling","crémant"}
ROSE_TERMS = {"rosé","rose"}

def _infer_wine_type(sub_comm: str) -> str | None:
    s = sub_comm.lower()
    if any(t in s for t in SPARKLING_TERMS): return "sparkling"
    if any(t in s for t in ROSE_TERMS): return "rosé"
    if any(t in s for t in RED_VARIETALS): return "red"
    if any(t in s for t in WHITE_VARIETALS): return "white"
    return None

@dataclass
class CsvRow:
    upc: str | None
    wine_name: str
    brand: str | None
    country: str | None
    varietal: str | None
    wine_type: str | None
    store_id: str | None
    store_name: str | None
    address: str | None
    zip_code: str | None
    bottle_size: str | None
    price: float | None
    last_scraped_at: str | None

def parse_row(raw: dict) -> CsvRow:
    name1 = raw.get("cust_frndly1_des", "").strip()
    name2 = raw.get("cust_frndly2_des", "").strip()
    wine_name = f"{name1} {name2}".strip() if name2 else name1
    sub_comm = raw.get("dsc_sub_comm", "").strip()
    price_str = raw.get("aip", "").strip()
    return CsvRow(
        upc=raw.get("id_upc", "").strip() or None,
        wine_name=wine_name,
        brand=raw.get("dsc_brand", "").strip() or None,
        country=raw.get("dsc_coo", "").strip() or None,
        varietal=sub_comm or None,
        wine_type=_infer_wine_type(sub_comm),
        store_id=raw.get("id_str", "").strip() or None,
        store_name=raw.get("cust_frndly_nm", "").strip() or None,
        address=raw.get("add_str", "").strip() or None,
        zip_code=raw.get("cd_zip", "").strip() or None,
        bottle_size=raw.get("sz_item", "").strip() or None,
        price=float(price_str) if price_str else None,
        last_scraped_at=raw.get("max_sales_date", "").strip() or None,
    )

def seed(csv_path: Path):
    client = get_service_client()
    rows: list[CsvRow] = []
    with open(csv_path, newline="", encoding="utf-8-sig") as f:
        for raw in csv.DictReader(f):
            rows.append(parse_row(raw))

    # Upsert wines
    wine_records = [
        {k: v for k, v in {
            "upc": r.upc, "name": r.wine_name, "brand": r.brand,
            "country": r.country, "varietal": r.varietal,
            "wine_type": r.wine_type, "bottle_size": r.bottle_size,
            "avg_price": r.price,
        }.items() if v is not None}
        for r in rows if r.wine_name
    ]
    client.table("wines").upsert(wine_records, on_conflict="upc").execute()

    # Fetch wine UPC → ID map
    result = client.table("wines").select("id,upc").execute()
    upc_to_id = {w["upc"]: w["id"] for w in result.data if w["upc"]}

    # Upsert retail inventory
    inv_records = []
    for r in rows:
        wine_id = upc_to_id.get(r.upc) if r.upc else None
        inv_records.append({k: v for k, v in {
            "wine_id": wine_id,
            "upc": r.upc,
            "retailer_name": "HEB",
            "store_id": r.store_id,
            "store_name": r.store_name,
            "address": r.address,
            "zip_code": r.zip_code,
            "price": r.price,
            "in_stock": True,
            "last_scraped_at": r.last_scraped_at,
        }.items() if v is not None})

    client.table("retail_inventory").upsert(
        inv_records, on_conflict="upc,store_id"
    ).execute()

    print(f"Seeded {len(wine_records)} wines, {len(inv_records)} inventory records")

if __name__ == "__main__":
    csv_path = Path(__file__).parents[2] / "data" / "seed" / "Fair_Oaks_Stores_Analysis.csv"
    seed(csv_path)
```

# backend/enrichment/grapeminds.py
import json
import subprocess
from dataclasses import dataclass, field
from config import settings

BASE_URL = "https://api.grapeminds.eu/public/v1"

@dataclass
class GrapeMindsWine:
    grapeminds_id: str
    display_name: str
    color: str | None = None
    producer_name: str | None = None
    region_name: str | None = None
    grapes: list[str] = field(default_factory=list)
    description: str | None = None
    description_long: str | None = None
    tasting_notes: str | None = None
    tasting_notes_long: str | None = None
    pairing: str | None = None
    pairing_long: str | None = None
    structure_profile: dict = field(default_factory=dict)
    is_fully_enriched: bool = False

@dataclass
class DrinkingPeriod:
    from_year: int | None = None
    to_year: int | None = None
    statement: str | None = None
    young: str | None = None
    ripe: str | None = None
    storage: str | None = None

class GrapeMindsClient:
    def __init__(self):
        self.api_key = settings.grapeminds_api_key

    def _call(self, path: str) -> dict:
        result = subprocess.run(
            ["curl", "-s",
             "-H", f"Authorization: Bearer {self.api_key}",
             "-H", "Accept-Language: en",
             f"{BASE_URL}{path}"],
            capture_output=True, text=True, timeout=15
        )
        return json.loads(result.stdout)

    def _parse_search(self, payload: dict) -> list[dict]:
        return payload.get("data", [])

    def _parse_detail(self, payload: dict) -> GrapeMindsWine:
        d = payload.get("data", {})
        fp = d.get("flavor_profile") or {}
        desc = d.get("description") or {}
        notes = d.get("tasting_notes") or {}
        pairing = d.get("pairing") or {}
        grapes = [g["name"] for g in (d.get("grapes") or [])]
        is_enriched = bool(fp or desc.get("text"))
        return GrapeMindsWine(
            grapeminds_id=str(d.get("id", "")),
            display_name=d.get("display_name", ""),
            color=d.get("color"),
            producer_name=(d.get("producer") or {}).get("name"),
            region_name=(d.get("region") or {}).get("name"),
            grapes=grapes,
            description=desc.get("text") if isinstance(desc, dict) else None,
            description_long=desc.get("text_long") if isinstance(desc, dict) else None,
            tasting_notes=notes.get("text") if isinstance(notes, dict) else None,
            tasting_notes_long=notes.get("text_long") if isinstance(notes, dict) else None,
            pairing=pairing.get("text") if isinstance(pairing, dict) else None,
            pairing_long=pairing.get("text_long") if isinstance(pairing, dict) else None,
            structure_profile=fp if isinstance(fp, dict) else {},
            is_fully_enriched=is_enriched,
        )

    def _parse_drinking_period(self, payload: dict) -> DrinkingPeriod | None:
        if payload.get("generating"):
            return None
        return DrinkingPeriod(
            from_year=payload.get("from"),
            to_year=payload.get("to"),
            statement=payload.get("statement"),
            young=payload.get("young"),
            ripe=payload.get("ripe"),
            storage=payload.get("storage"),
        )

    def search(self, wine_name: str, limit: int = 5) -> list[dict]:
        q = wine_name.replace(" ", "+")
        payload = self._call(f"/wines/search?q={q}&limit={limit}")
        return self._parse_search(payload)

    def get_wine(self, grapeminds_id: int) -> GrapeMindsWine:
        payload = self._call(f"/wines/{grapeminds_id}")
        return self._parse_detail(payload)

    def get_drinking_period(self, grapeminds_id: int) -> DrinkingPeriod | None:
        payload = self._call(f"/drinking-periods/{grapeminds_id}")
        return self._parse_drinking_period(payload)

# backend/enrichment/pipeline.py
import asyncio
from dataclasses import dataclass, field
from db import get_service_client
from enrichment.grapeminds import GrapeMindsClient

@dataclass
class EnrichmentResult:
    wine_id: str
    grapeminds_id: str | None = None
    description: str | None = None
    description_long: str | None = None
    tasting_notes: str | None = None
    tasting_notes_long: str | None = None
    pairing: str | None = None
    pairing_long: str | None = None
    structure_profile: dict = field(default_factory=dict)
    grapes: list[str] = field(default_factory=list)
    drinking_window_start: int | None = None
    drinking_window_end: int | None = None
    drinking_window_young: str | None = None
    drinking_window_ripe: str | None = None
    drinking_window_storage: str | None = None
    source: str = "grapeminds"
    needs_refetch: bool = False

async def enrich_wine(wine_row: dict) -> EnrichmentResult:
    result = EnrichmentResult(wine_id=wine_row["id"])
    gm = GrapeMindsClient()

    search_hits = gm.search(wine_row["name"], limit=3)
    if not search_hits:
        result.source = "pending"
        result.needs_refetch = True
        return result

    gm_id = search_hits[0]["id"]
    wine_data = gm.get_wine(gm_id)
    drinking = gm.get_drinking_period(gm_id)

    result.grapeminds_id = wine_data.grapeminds_id
    result.description = wine_data.description
    result.description_long = wine_data.description_long
    result.tasting_notes = wine_data.tasting_notes
    result.tasting_notes_long = wine_data.tasting_notes_long
    result.pairing = wine_data.pairing
    result.pairing_long = wine_data.pairing_long
    result.structure_profile = wine_data.structure_profile
    result.grapes = wine_data.grapes
    result.needs_refetch = not wine_data.is_fully_enriched

    if drinking:
        result.drinking_window_start = drinking.from_year
        result.drinking_window_end = drinking.to_year
        result.drinking_window_young = drinking.young
        result.drinking_window_ripe = drinking.ripe
        result.drinking_window_storage = drinking.storage

    return result

async def persist_enrichment(result: EnrichmentResult):
    client = get_service_client()
    record = {k: v for k, v in {
        "wine_id": result.wine_id,
        "grapeminds_id": result.grapeminds_id,
        "description": result.description,
        "description_long": result.description_long,
        "tasting_notes": result.tasting_notes,
        "tasting_notes_long": result.tasting_notes_long,
        "pairing": result.pairing,
        "pairing_long": result.pairing_long,
        "structure_profile": result.structure_profile,
        "flavor_profile": result.grapes,
        "drinking_window_start": result.drinking_window_start,
        "drinking_window_end": result.drinking_window_end,
        "drinking_window_young": result.drinking_window_young,
        "drinking_window_ripe": result.drinking_window_ripe,
        "drinking_window_storage": result.drinking_window_storage,
        "source": result.source,
        "grapeminds_enriched_at": None if result.needs_refetch else "NOW()",
        "enriched_at": None if result.needs_refetch else "NOW()",
    }.items() if v is not None}
    client.table("wine_details").upsert(record, on_conflict="wine_id").execute()

async def enrich_and_persist(wine_row: dict, refetch_delay: int = 60):
    result = await enrich_wine(wine_row)
    await persist_enrichment(result)
    if result.needs_refetch:
        # GrapeMinds is generating content async — re-fetch after delay
        await asyncio.sleep(refetch_delay)
        result2 = await enrich_wine(wine_row)
        await persist_enrichment(result2)
```

# backend/scrapers/base.py
import uuid
from abc import ABC, abstractmethod
from dataclasses import dataclass
from datetime import datetime, timezone
from db import get_service_client

@dataclass
class RetailInventoryItem:
    wine_name: str
    upc: str | None
    price: float | None
    store_name: str
    store_id: str | None
    address: str | None
    city: str | None
    state: str | None
    zip_code: str
    in_stock: bool
    retailer_name: str

class BaseScraper(ABC):
    def __init__(self):
        self.supabase = get_service_client()

    @abstractmethod
    async def search_by_zip(self, zip_code: str) -> list[RetailInventoryItem]: ...

    @abstractmethod
    async def search_by_wine(self, wine_name: str, zip_code: str) -> list[RetailInventoryItem]: ...

    async def upsert_inventory(self, items: list[RetailInventoryItem], run_id: str):
        records = [{k: v for k, v in {
            "upc": item.upc,
            "retailer_name": item.retailer_name,
            "store_id": item.store_id,
            "store_name": item.store_name,
            "address": item.address,
            "city": item.city,
            "state": item.state,
            "zip_code": item.zip_code,
            "price": item.price,
            "in_stock": item.in_stock,
            "last_scraped_at": datetime.now(timezone.utc).isoformat(),
        }.items() if v is not None} for item in items]
        self.supabase.table("retail_inventory") \
            .upsert(records, on_conflict="upc,store_id") \
            .execute()

    async def run(self, zip_codes: list[str]) -> int:
        run_id = str(uuid.uuid4())
        self.supabase.table("scraper_runs").insert({
            "id": run_id, "retailer_name": self.__class__.__name__, "status": "running"
        }).execute()
        total = 0
        try:
            for zip_code in zip_codes:
                items = await self.search_by_zip(zip_code)
                await self.upsert_inventory(items, run_id)
                total += len(items)
            self.supabase.table("scraper_runs").update({
                "status": "success", "records_updated": total,
                "completed_at": datetime.now(timezone.utc).isoformat(),
            }).eq("id", run_id).execute()
        except Exception as e:
            self.supabase.table("scraper_runs").update({
                "status": "failed", "error_message": str(e),
                "completed_at": datetime.now(timezone.utc).isoformat(),
            }).eq("id", run_id).execute()
            raise
        return total
```

# backend/scrapers/heb.py
import asyncio, random, re
from playwright.async_api import async_playwright
from bs4 import BeautifulSoup
from scrapers.base import BaseScraper, RetailInventoryItem

class HEBScraper(BaseScraper):
    BASE_URL = "https://www.heb.com"

    def _parse_price(self, text: str) -> float | None:
        match = re.search(r"[\d]+\.?\d*", text.replace(",", ""))
        return float(match.group()) if match else None

    def _parse_products(self, html: str, zip_code: str,
                        store_name: str, store_id: str) -> list[RetailInventoryItem]:
        soup = BeautifulSoup(html, "html.parser")
        items = []
        for tile in soup.select(".product-tile"):
            name_el = tile.select_one(".product-tile__description")
            price_el = tile.select_one(".product-tile__price")
            if not name_el:
                continue
            items.append(RetailInventoryItem(
                wine_name=name_el.text.strip(),
                upc=None,
                price=self._parse_price(price_el.text) if price_el else None,
                store_name=store_name,
                store_id=store_id,
                address=None, city=None, state="TX",
                zip_code=zip_code, in_stock=True, retailer_name="HEB",
            ))
        return items

    async def _fetch(self, url: str) -> str:

```python
# backend/recommender/candidate_retrieval.py
from dataclasses import dataclass, field
from db import get_service_client

AVOID_STRUCTURE_MAP = {
    "sweet": ("sweetness", 6),    # sweetness > 6 = avoid if "sweet" in avoid list
    "light bodied": ("body", 4),  # body < 4 = avoid if "light bodied" in avoid list
    "high tannin": ("tannins", 7),
}

PREFER_STRUCTURE_MAP = {
    "bold": ("body", 7),
    "full bodied": ("body", 7),
    "tannic": ("tannins", 7),
    "crisp": ("acidity", 6),
    "dry": ("sweetness", 3),
}

@dataclass
class UserPreferences:
    zip_code: str
    budget_min: float
    budget_max: float
    style_preferences: list[str]
    avoid: list[str]
    wine_type: str | None = None

def score_candidate(price: float, structure: dict,
                    wine_type: str | None, prefs: UserPreferences) -> float:
    if price < prefs.budget_min or price > prefs.budget_max:
        return 0.0

    score = 0.5  # base score for in-budget

    # Penalize avoided styles via structure
    for avoid_term in prefs.avoid:
        key, threshold = AVOID_STRUCTURE_MAP.get(avoid_term.lower(), (None, None))
        if key and key in structure:
            if avoid_term.lower() == "sweet" and structure[key] > threshold:
                score -= 0.3
            elif avoid_term.lower() == "light bodied" and structure[key] < threshold:
                score -= 0.3

    # Reward preferred styles via structure
    hits = 0
    for pref in prefs.style_preferences:
        key, threshold = PREFER_STRUCTURE_MAP.get(pref.lower(), (None, None))
        if key and key in structure and structure[key] >= threshold:
            hits += 1
    if prefs.style_preferences:
        score += 0.4 * (hits / len(prefs.style_preferences))

    # Wine type match bonus
    if prefs.wine_type and wine_type == prefs.wine_type:
        score += 0.1

    return max(0.0, min(1.0, score))

def get_candidates(prefs: UserPreferences, limit: int = 15) -> list[dict]:
    client = get_service_client()
    result = client.table("retail_inventory") \
        .select("""
            id, price, store_name, store_id, address, zip_code, in_stock,
            wines(id, name, brand, varietal, region, country, wine_type, avg_price,
                  wine_details(description, tasting_notes, flavor_profile,
                               structure_profile, drinking_window_start,
                               drinking_window_end, grapeminds_id))
        """) \
        .eq("zip_code", prefs.zip_code) \
        .gte("price", prefs.budget_min) \
        .lte("price", prefs.budget_max) \
        .eq("in_stock", True) \
        .execute()

    scored = []
    for row in result.data:
        wine = row.get("wines") or {}
        detail = (wine.get("wine_details") or [{}])[0]
        structure = detail.get("structure_profile") or {}
        score = score_candidate(
            price=row["price"],
            structure=structure,
            wine_type=wine.get("wine_type"),
            prefs=prefs,
        )
        if score > 0:
            scored.append((score, row))

    scored.sort(key=lambda x: x[0], reverse=True)
    return [row for _, row in scored[:limit]]
```

```python
# backend/recommender/engine.py
import anthropic
from config import settings

SYSTEM_PROMPT = """You are an expert wine sommelier named Terroir. Given the user's preferences
and a list of wines currently available at stores near them, recommend 3–5 wines with
substantive explanations. For each wine:
- Explain why it matches their stated preferences (be specific about flavor, structure, finish)
- Mention the region and what makes it distinctive
- Note the drinking window if known and whether it's drinking well now
- Include the store name, address, and price
- Be opinionated — if one wine clearly stands out, say so

Do not lead with food pairings. Write like a knowledgeable friend, not a textbook.
Use short paragraphs. Avoid vague descriptors without context."""

def build_candidate_context(candidates: list[dict]) -> str:
    lines = []
    for c in candidates:
        wine = c.get("wines") or {}
        details = (wine.get("wine_details") or [{}])[0]
        structure = details.get("structure_profile") or {}
        dw_start = details.get("drinking_window_start")
        dw_end = details.get("drinking_window_end")
        drinking = f"{dw_start}–{dw_end} years" if dw_start and dw_end else "unknown"
        lines.append(
            f"- {wine.get('name', 'Unknown')} by {wine.get('brand', 'Unknown')}\n"
            f"  Region: {wine.get('region', 'Unknown')} | Type: {wine.get('wine_type', '')}\n"
            f"  Price: ${c.get('price', 0):.2f} at {c.get('store_name', '')} ({c.get('address', '')})\n"
            f"  Tasting notes: {details.get('tasting_notes', 'N/A')}\n"
            f"  Structure: {structure}\n"
            f"  Drinking window: {drinking}"
        )
    return "\n\n".join(lines)

class RecommendationEngine:
    def __init__(self):
        self.client = anthropic.Anthropic(api_key=settings.anthropic_api_key)

    def recommend(self, candidates: list[dict], conversation_history: list[dict],
                  user_message: str) -> str:
        context = build_candidate_context(candidates)
        user_content = f"{user_message}\n\nAvailable wines near you:\n{context}"
        messages = conversation_history + [{"role": "user", "content": user_content}]
        response = self.client.messages.create(
            model="claude-sonnet-4-6",
            max_tokens=2048,
            system=SYSTEM_PROMPT,
            messages=messages,
        )
        return response.content[0].text
```
